h1. Installation:

# Copy memcached.so to /Applications/MAMP/bin/php/php5.3.x/lib/php/extensions/no-debug-non-zts-20090626 (Channge 5.3.x to your PHP version)
# Add extension="memcached.so" to the end of php.ini (Open MAMP click on File → Edit Template → PHP → PHP 5.3.x php.ini) (Channge 5.3.x to your PHP version)
# Restart MAMP.

h2. Dependencies:
# memcached (brew install memcached)
# libmemcached (brew install libmemcached)
