h1. Installation:

# Copy memcached.so to /Applications/MAMP/bin/php/php5.3.6/lib/php/extensions/no-debug-non-zts-20090626
# Add extension="memcached.so" to the end of php.ini (Open MAMP click on File → Edit Template → PHP → PHP 5.3.6 php.ini)
# Restart MAMP.

h2. Dependencies
	memcached (brew install memcached)
	libmemcached (brew install libmemcached)
